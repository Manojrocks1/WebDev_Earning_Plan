   // ===================================================================
// 1. THIRD-PARTY LIBRARY INITIALIZATION
// ===================================================================

// Initialize AOS (Animate On Scroll) library
// This function scans the document for elements with 'data-aos' attributes 
// and triggers animations as they come into view.
AOS.init({ 
    // Set animation duration to 1500ms (1.5 seconds)
    duration: 1500, 
    // Animation only happens once per element (it won't repeat on scroll up/down)
    once: true 
});

// ===================================================================
// 2. MOBILE SIDEBAR/MENU CONTROLS
// ===================================================================

// Get references to the DOM elements for the mobile menu functionality
const menuBtn = document.getElementById('menu-btn');      // The Hamburger icon button (â˜°)
const sidebar = document.getElementById('sidebar');      // The slide-in mobile navigation panel
const overlay = document.getElementById('overlay');      // The semi-transparent background overlay
const closeBtn = document.getElementById('close-btn');    // The close button (Ã—) inside the sidebar

// Functionality to open the sidebar when the hamburger button is clicked
menuBtn.onclick = () => {
    // Remove the 'translate-x-full' class (Tailwind) to slide the sidebar into view
    sidebar.classList.remove('translate-x-full');
    // Remove the 'hidden' class to make the background overlay visible
    overlay.classList.remove('hidden');
};

// Functionality to close the sidebar when the close button is clicked
closeBtn.onclick = () => {
    // Add the 'translate-x-full' class to slide the sidebar out of view
    sidebar.classList.add('translate-x-full');
    // Add the 'hidden' class to hide the background overlay
    overlay.classList.add('hidden');
};

// Functionality to close the sidebar when the overlay (clicking outside the menu) is clicked
overlay.onclick = () => {
    // Add the 'translate-x-full' class to slide the sidebar out
    sidebar.classList.add('translate-x-full');
    // Add the 'hidden' class to hide the background overlay
    overlay.classList.add('hidden');
};

// Note: Clicking on a link inside the sidebar should also close it for better UX. 
// This is typically handled by adding the close logic to all sidebar links.

// ===================================================================
// 3. SCROLL-TO-TOP BUTTON VISIBILITY
// ===================================================================

// Get reference to the Scroll-to-Top button element
const scrollTopBtn = document.getElementById('scrollTopBtn');

// Add an event listener to the window object to monitor scrolling
window.addEventListener('scroll', () => {
    // Check if the vertical scroll position (window.scrollY) is greater than 300 pixels
    if (window.scrollY > 300) {
        // If scrolled down enough, remove 'hidden' class to show the button
        scrollTopBtn.classList.remove('hidden');
    } else {
        // If near the top of the page, add 'hidden' class to hide the button
        scrollTopBtn.classList.add('hidden');
    }
});

// Functionality to smoothly scroll back to the top of the page when the button is clicked
scrollTopBtn.onclick = () => window.scrollTo({ 
    top: 0, 
    // Use 'smooth' behavior for a gentle, animated scroll
    behavior: 'smooth' 
});

// ===================================================================
// 4. FEATURE: ACTIVE NAVIGATION LINK HIGHLIGHTING (Intersection Observer)
// ===================================================================

// Get all navigation links in the header and all section elements (anchors)
const navLinks = document.querySelectorAll('header nav a');
const sections = document.querySelectorAll('section');

// Helper function to remove the active highlight class from all links
function removeActiveClasses() {
    navLinks.forEach(link => link.classList.remove('nav-active'));
}

// Configuration options for the Intersection Observer
const observerOptions = {
    root: null, // The viewport is the container (root)
    // Defines the area around the root. 
    // '-20% 0px' means the section is considered active when its top edge 
    // is 20% down from the top of the viewport.
    rootMargin: '0px 0px -20% 0px', 
    // Threshold is 0.1: Triggers the callback when 10% of the target section is visible
    threshold: 0.1 
};

// Create the Intersection Observer instance
const sectionObserver = new IntersectionObserver((entries, observer) => {
    // Iterate over all observed elements that had a change in intersection status
    entries.forEach(entry => {
        // Check if the section is currently intersecting (visible within the rootMargin)
        if (entry.isIntersecting) {
            removeActiveClasses(); // Step 1: Clear all current active highlights

            // Step 2: Get the ID of the visible section (e.g., 'home', 'about', 'services')
            const currentSectionId = entry.target.id; 
            
            // Step 3: Find the corresponding navigation link using its href attribute
            // (e.g., looks for 'header nav a[href="#home"]')
            const targetLink = document.querySelector(`header nav a[href="#${currentSectionId}"]`);
            
            // Step 4: Add the custom 'nav-active' class to highlight the link
            if (targetLink) {
                targetLink.classList.add('nav-active');
            }
        }
    });
}, observerOptions);

// Start observing all section elements in the document
sections.forEach(section => {
    sectionObserver.observe(section);
});
